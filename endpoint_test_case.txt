## ENDPOINT HEALTH CHECK — Scenarios

Scenario 1 — Basic HTTPS Health Check (baseline)

provider "aws" {
  region = "ap-south-1"
}

module "test_endpoint_basic" {
  source = "../../"

  name = "endpoint-basic"

  type = "HTTPS"

  fqdn = "example.com"
  port = 443

  resource_path = "/"

  request_interval  = 30
  failure_threshold = 3
}

output "health_check_id" {
  value = module.test_endpoint_basic.health_check_id
}

Scenario 4 — TCP Health Check (Database / Internal Service)

module "endpoint_tcp_healthcheck" {
  source = "../../"

  name = "prod-db-healthcheck"

  type = "TCP"

  fqdn = "db.example.com"   # Replace with real endpoint
  port = 3306               # MySQL (can be 5432 for Postgres, 6379 for Redis)

  request_interval  = 30
  failure_threshold = 3

  tags = {
    Environment = "prod"
    Service     = "database"
  }
}

multi region test 


module "endpoint_multi_region_healthcheck" {
  source = "../../"

  name = "multi-region-healthcheck"

  type = "HTTPS"

  fqdn = "example.com"
  port = 443

  resource_path = "/health"

  request_interval  = 10     # ✅ faster detection
  failure_threshold = 5      # ✅ avoid false positives

  measure_latency = true     # ✅ latency tracking

  regions = [
    "us-east-1",
    "us-west-2",
    "eu-west-1",
    "ap-south-1"
  ]

  tags = {
    Environment = "prod"
    Strategy    = "multi-region-ha"
  }
}
